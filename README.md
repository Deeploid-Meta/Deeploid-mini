Rscript dada2_OTU_full.R -p /Users/d.voronkina/Desktop/Deeploid/deeploid-mini-cli/data -1 '*_R1.fastq.gz' -2 '*_R2.fastq.gz' -o /Users/d.voronkina/Desktop/Deeploid/deeploid-mini-cli/output
